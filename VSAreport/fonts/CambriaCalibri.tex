\newcommand\lightseries{\addfontfeatures{FakeBold=0.5}\selectfont}
\newcommand\thinseries{\addfontfeatures{FakeBold=0.2}\selectfont}
\newcommand\blackseries{\addfontfeatures{FakeBold=2.0}\selectfont}
\newcommand\mediumseries{\addfontfeatures{FakeBold=1.0}\selectfont}
\newcommand\condseries{\addfontfeatures{Scale=0.85}\selectfont}
\newcommand\condlightseries{\addfontfeatures{Scale=0.85,FakeBold=0.5}\selectfont}
\newcommand\condboldseries{\addfontfeatures{Scale=0.85,FakeBold=2.0}\selectfont}
